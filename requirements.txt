# requirements.txt - SIGS (Sistema Integrado de Gerenciamento de Senhas)
# =============================================================
# Instalação:
#     pip install -r requirements.txt
#
# Observação importante:
#   As bibliotecas pywin32 (win32print, win32ui, win32con) e Pillow
#   somente são necessárias na máquina Windows onde o sistema fará a
#   impressão física dos tickets. Para desenvolvimento/testes da parte
#   web em outros sistemas operacionais, a aplicação Flask funciona
#   normalmente sem elas (a impressão simplesmente não estará
#   disponível, retornando um erro amigável ao usuário).
# =============================================================

Flask==3.0.3
Werkzeug==3.0.3

# Impressão nativa do Windows (obrigatório em produção no Windows).
pywin32>=306; sys_platform == "win32"

# Usado por printer.py para desenhar o logotipo no ticket impresso.
Pillow==10.4.0

# Geração de relatórios em Excel (.xlsx).
openpyxl==3.1.5

# Geração de relatórios em PDF (apenas relatórios gerenciais, nunca o
# ticket de senha, que é sempre impresso via GDI/pywin32).
reportlab==4.2.5

# Servidor WSGI recomendado para produção no Windows (ver README.md).
waitress==3.0.0
